26 class StreamComparison( filter : Iterator[PullType] => Iterator[PullType] = identity)( implicit ic : XmlComparison[XmlItem], ec : XmlComparison[Elem], endElemQNameEqual : Equal[QName], qnameTokenComparison : Option[(ComparisonContext, String, String) => Boolean]) extends XmlComparison[StreamComparable[_]] {
27   def compare( calculate : Boolean, ocontext : ComparisonContext , lefti : StreamComparable[_], righti : StreamComparable[_] ) : Option[(XmlDifference[_], ComparisonContext)] = {
28 
29     var res : Option[(XmlDifference[_], ComparisonContext)] = None
30     val joined = filter(lefti.underlyingIterator).zip(filter(righti.underlyingIterator))
31     var context = ocontext
32     var streamPosition = 0
33 
34     while( res.isEmpty && joined.hasNext) {
35       streamPosition += 1
36 
37       joined.next match {
38 	case (Left(x : Elem), Left(y : Elem)) =>
39 	  if (calculate || qnameTokenComparison.isDefined) { // also needed for qname handling
40 	    context = context.startElems(x, y)
41 	  }
42 	  res = ec.compare(calculate, context, x, y)
43 	case (Left(x : XmlItem), Left(y : XmlItem)) =>
44 	  res = ic.compare(calculate, context, x, y)
45 	case (Right(x : EndElem), Right(y  : EndElem)) =>
46 	  res = 
47 	    if (!endElemQNameEqual.equal(x.name, y.name)) {
48 	      if (calculate)
49 		Some((EndElemNameDifference(x, y), context))
50 	      else 
51 		noCalculation
52 	    } else None
53 	  if (calculate) { // also needed for qname handling
54 	    context = context.endElem
55 	  }
56 	case (left, right) => 
57 	  res = 
58 	    if (calculate || qnameTokenComparison.isDefined)
59 	      Some((DifferentTypes(left, right), context)) 
60 	    else
61 	      noCalculation
62       }
63     }
64     if (calculate && res.isDefined) {
65       // unpack, repack
66       val Some((diff, rcontext)) = res
67       context = rcontext.withPosition(streamPosition)
