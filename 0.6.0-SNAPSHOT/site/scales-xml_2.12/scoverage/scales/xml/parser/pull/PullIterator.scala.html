115   protected[xml] def pumpEvent: PullType = {
116     if (!parser.hasNext) return null
117 
118     var nextEvent = XMLStreamConstants.END_DOCUMENT
119     try {
120       val (event, num, odepth, oprolog) = PullUtils.pumpEvent(parser, strategy, token, vprolog, depth){_ => pumpEvent}
121       nextEvent = num
122       depth = odepth
123       vprolog = oprolog
124       event
125     } finally {
126       // should we close it? 
127       if (nextEvent == XMLStreamConstants.END_DOCUMENT) {

211   def pumpEvent[Token <: OptimisationToken]( parser: XMLStreamReader, strategy : MemoryOptimisationStrategy[Token], token : Token, prolog : Prolog, idepth : Int )(otherEventHandler : Int => PullType) : (PullType, Int, Int, Prolog) = {
