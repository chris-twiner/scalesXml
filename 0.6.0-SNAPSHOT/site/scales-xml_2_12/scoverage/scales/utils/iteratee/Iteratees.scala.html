257     def step(l : List[ResumableIter[E,A]])(s: Input[E]): ResumableIterList[E,A] = {
258       iteratee(
259         s(el = e => {
260           var res : List[A] = Nil
261           var newl : List[ ResumableIter[E,A] ] = Nil
262 
263           @inline def add( k : (Input[E]) => Iteratee[E,(A, Iteratee[E, _])]) {
264             val d = k(Element(e))
265             newl = d :: newl
266             d foldT[Unit] (
267               done = (x, _) => res = x._1 :: res,
268               cont = _ => ()
269             )
270           }
271 
272           val i = l.iterator
273           while(i.hasNext) {
274             i.next.foldT(
275               // safety first
276               done = (e1, y) =>
277                 // are we EOF, in which case remove
278                 if (isEOF(iteratee(Done(e1,y))))
279                   Nil
280                 else {
281                   if (isEmpty(iteratee(Done(e1,y))))
282                     // feed back the continuation
283                     // this is where we hope that the users don't
284                     // break on purpose :-()
285                     e1._2.foldT(
286                       cont = k => add(k.asInstanceOf[(Input[E]) => Iteratee[E,(A, Iteratee[E, _])]]),
287                       done = (x,y) => error("Continuation can only be a Cont")
288                     )
289                   else
290                     error("Can only handle EOF or Empty for Done")
291 
292                 },
293               cont = i => add(i)
294             )
295           }
296 
297           if (res.isEmpty)
298             Cont(step(newl))
299           else
300             Done((res, iteratee(Cont(step(newl)))), Empty[E])
301 
302         },
303         empty = Cont(step(l)),
304         eof = Done((Nil, iteratee(Cont(step(l)))), Eof[E])
305         )
306       )
307     }
308 
309     iteratee( Cont(step(originalList)) )
310   }
