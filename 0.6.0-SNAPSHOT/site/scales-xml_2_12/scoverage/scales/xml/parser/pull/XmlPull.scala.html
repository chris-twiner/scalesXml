78     val token = strategy.createToken(Xml10, IsFromParser)
79     // start with nothing
80     val buf = new impl.TreeProxies()
81 
82     while( pull.hasNext ){

86 	case Left( e : Elem ) =>

88 	case Right(endElem) => 
89 	  strategy.elementEnd(buf, token)

91     }
