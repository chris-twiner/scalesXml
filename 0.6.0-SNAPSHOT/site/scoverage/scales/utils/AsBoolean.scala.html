3 /**
4  * Provides the conversion type class AsBoolean, and simple matchers (extractor builders) to
5  * use them.  Used by the xpath boolean function and xmltree matching
6  */ 
7 trait AsBooleanTrait {
8   type AsBoolean[T] = Function1[T, Boolean]
9   def boolean[T : AsBoolean]( it : T ) : Boolean = implicitly[AsBoolean[T]].apply(it)
10   
11   class BooleanMatcher[X, T : AsBoolean](eval  : (X) => T) {
12     def unapply( x : X ) : Option[T] = {
13       val res = eval(x)
14       if (!implicitly[AsBoolean[T]].apply(res)) None
15       else Some(res)
16     }
17   }
18 
19   class BooleanAndTMatcher[X, T : AsBoolean](eval : (X) => T) {
20     def unapply( x : X ) : Option[(T, X)] = {
21       val res = eval(x)
22       if (!implicitly[AsBoolean[T]].apply(res)) None
23       else Some((res, x))
24     }
25   }
26 
27   /**
28    * Calls eval to return a T t'.  This T is then evaluated with the AsBoolean type class.
29    *
30    * When evaluated to true t' is returned.
31    * val Matcher = booleanMatcher( (x) => T ); val Matcher(t) = x
32    */ 
33   def booleanMatcher[X, T : AsBoolean]( eval : (X) => T ) = new BooleanMatcher[X, T](eval)
34 
35   /**
36    * Calls eval to return a T t'.  This T is then evaluated with the AsBoolean type class.
37    *
38    * When evaluated to true (t',x) is returned.
39    * val Matcher = booleanMatcher( (x) => T ); val Matcher(t,x) = x
40    */
41   def booleanAndTMatcher[X, T : AsBoolean]( eval : (X) => T ) = new BooleanAndTMatcher[X, T](eval)
42 }
