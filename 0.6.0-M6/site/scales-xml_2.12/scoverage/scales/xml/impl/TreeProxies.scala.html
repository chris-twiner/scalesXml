33   /**
34    * Strips a path from the current position to the top of tree.  The
35    * existing cached trees are then effectively re-created without the current position (one depth less)
36    */ 
37   def proxyPath() : XmlPath = {
38     var d = _depth
39     val l = _current
40 
41     val te = l.elem
42     val tcc = l.builder.result
43      
44     var elems = Array.ofDim[Elem](if (d < 0) 0 else d)
45     while( d > 0 ) {
46       d -= 1
47 
48       elems(d) = _proxies(d).elem
49     }
50 
51     reuse
52 
53     var p = noXmlPath
54 
55     // add them back
56     d = 0
57     while( d < elems.length ){
58       beginSub(elems(d), XmlBuilder())
59       // add child to path
60       p = addAndFocus(p, elems(d))
61       d += 1
62     }
63     p = addAndFocus(p, te, tcc)
64     
65     p
66   }

68   /**

72   def proxyRemoveAndUp() : TreeProxies = {

144     _depth += 1

157     } else {
158       _current = _proxies(_depth)
159       _current.setElem(elem)
160       _current.builder.clear() // don't create a new one
161     }
162   }
