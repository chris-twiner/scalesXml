23 trait XmlEquals {
24 
25   /**
26    * Allows comparison of Text nodes or Attribute values that contain QNames, i.e. prefix:value.  In order to perform this comparison both the left and right sides must have a NamespaceContext.
27    */ 
28   def qnamesEqual(context : ComparisonContext, str : String, str2 : String) = {
29     // split both, if one has and the other not, then its false anyway
30     val sp1 = str.split(":")
31     val sp2 = str2.split(":")
32     if (sp1.size == 2 && sp2.size == 2) {
33       sp1(1) == sp2(1) && { // values match
34 	// look up prefixes
35 	(for{ lnc <- context.leftNamespaceContext
36 	     rnc <- context.rightNamespaceContext
37 	     lns <- lnc.mappings.get(sp1(0))
38 	     rns <- rnc.mappings.get(sp2(0))
39 	   } yield lns == rns
40         ).
41 	getOrElse(false)
42       }
43     } else 
44       str == str2
45   }
46 
47   /**
48    * Perform an actual token comparison (Text/CData and Attribute value relevant).
49    */ 
50   def compareTokens( context : ComparisonContext, qnameTokenComparison : Option[(ComparisonContext, String, String) => Boolean], str : String, str2 : String) =
51     qnameTokenComparison.
52       map( f =>
53 	f(context, str, str2)
54 	).
55       getOrElse(str == str2)
56 
57 
58   /**
59    * Compare the xml object via the available XmlDifference type class
60    */ 
61   def compare[T : XmlComparison]( left : T, right : T) : Option[(XmlDifference[_], ComparisonContext)] =
62     implicitly[XmlComparison[T]].compare(true, ComparisonContext(), left, right)
