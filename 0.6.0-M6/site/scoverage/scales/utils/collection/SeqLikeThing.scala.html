6 /**
7  * 2.13 removes seqlike etc. so we need to behave like it, it also deprecates canbuild fromm.  This interface represents the operations on a seq used by scales
8  *
9  * @tparam X[_] a seq like container
10  */
11 trait SeqLikeThing[Repr, A, X[_]] {
12 
13   implicit def cbf: CanBuildFrom[Repr, A, X[A]]
14 
15   @inline def apply(x: X[A])(i: Int): A = seq(x).apply(i)
16 
17   @inline def seq(x: X[A]): Seq[A]
18 
19   @inline def iterator(x: X[A]): Iterator[A] = seq(x).iterator
20 
21   @inline def updated(x: X[A])(index: Int, elem: A): X[A] =
22     wrap(seq(x).updated(index, elem))
23 
24   @inline def splitAt(x: X[A])(index: Int): (X[A],X[A]) = {
25     val (b, a) = seq(x: X[A]).splitAt(index)
26     (wrap(b), wrap(a))
27   }
28 
29   @inline def length(x: X[A]): Int = seq(x: X[A]).length
30 
31   @inline def ++(x: X[A])(itr: GenTraversableOnce[A]): X[A] =
32     wrap(seq(x: X[A]) ++ itr)
33 
34   @inline def wrap(s: Seq[A]): X[A]
35 
36   @inline def :+(x: X[A])(elem: A): X[A] =
37     wrap(seq(x: X[A]).:+(elem))
38 
39   @inline def dropRight(x: X[A])(n: Int): X[A] =
40     wrap(seq(x: X[A]).dropRight(n))
41 
42   @inline def filter(x: X[A])(p: A => Boolean): X[A] =
43     wrap(seq(x: X[A]).filter(p))
44 
45   @inline def filterNot(x: X[A])(p: A => Boolean): X[A] =
46     wrap(seq(x: X[A]).filterNot(p))
47 
48   @inline def builder: mutable.Builder[A, X[A]]
49 }
