10 /**
11  * QNames together with a tree structure form the basis of XML, what type of QName is available depends on Attribute (either no namespace or prefxied - fully qualified) or Elem (attribute's options and non prefixed but still qualified)
12  */ 
13 sealed trait QName {
14 
15   val local : String
16 
17   /**
18    * There is always a namespace, but it may be the noNamespace namespace
19    */ 
20   def namespace : UnderlyingNamespace
21   def prefix : Option[String]
22   
23   /**
24    * When the version is 1.1 it cannot be serialized to a 1.0 doc.
25    * This indicates with which version it is compatible
26    */ // to get here it must be valid so we only need to test for 1.0
27   def qNameVersion : XmlVersion = 
28     if (validLocalName(local)(Xml10) && 
29 	// treat non prefixed as Xml10 prefixes as they are compatible with both
30 	(prefix.map{p => validXmlPrefix(p)(Xml10)}.getOrElse(true)) &&
31 	((namespace eq EmptyNamespace)  ||
32 	  validXmlNamespace(namespace.uri)(Xml10) ))
33       Xml10
34     else 
35       Xml11
36 
37   def hasPrefix = prefix.isDefined
38 
39   /**
40    * local or prefix:local Namespaces QName, also XPath 2.0 functions version of a qualified name
41    */
42   def qName = 
43     if (prefix.isDefined) 
44       prefix.get + ":" + local
45     else local
46   
47   /**
48    * QName, pre:{namespace}localName JClark representation with prefix: on the front if it has a prefix.  CTw's own, for when you just really want that prefix.
49    */
50   def pqName = if (prefix.isDefined) prefix.get +":" + qualifiedName else qualifiedName
51 
52   /**
53    * QName, {namespace}localName JClark representation, as per everywhere else than w3c
54    */
55   def qualifiedName = "{" + namespace.uri + "}" + local
56 
57   override def toString = qualifiedName
58 
59   /** 
60     * Will match namespace and local with other AND prefix if available
61     */
62   def ====( other : QName) = 
63     if (this eq other) true
64     else
65     if (this =:= other) {
66       val hp = hasPrefix
67       val ohp = other.hasPrefix
68 
69       if (hp && ohp)
70     	prefix.get == other.prefix.get
71       else 
72 	(hp == ohp) // both false
73     } else false   
74 
75   /** 
76    * Will match namespace and local with other but not prefix ==== also does prefix
77    */
78   def =:=( other : QName) = 
79     (other eq this) ||
80     (local == other.local && namespace.uri == other.namespace.uri)
81   
82   /**
83    * Matches using =:= and does not work with === or prefixes
84    */ 
85   override def equals( other : Any ) = other match {
86     case oq : QName => this =:= oq
87     case _ => false
88   }
89 
90   override def hashCode() : Int = {
91     var hs = 1
92     hs = (hs * 31) + prefix.map(_.hashCode).getOrElse(1) // None is good here too
93     hs = (hs * 31) + local.hashCode
94     hs = (hs * 31) + namespace.hashCode
95     hs    
96   }
97 }
