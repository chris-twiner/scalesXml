44   def laddNS(entry: (String, String), mappings: Map[String, String], declMap: Map[String, String]) = {
45     val mapping = mappings.get(entry._1)
46 
47     if (mapping.isDefined) {
48       // currently exists
49       if (mapping.get == entry._2) {
50         // do nothing
51         (mappings, declMap)
52       } else {
53         // replace and redefine
54         (mappings.updated(entry._1, entry._2),
55           declMap + entry)
56       }
57     } else {
58       // map and define
59       (mappings + (entry._1 -> entry._2),
60         declMap + entry)
61     }
62   }

64   def doElement(x: Elem, currentMappings : Map[String, String]) = {
65     import ScalesXml._
66 
67     // for each of our namespace mappings that differ (or are not there)
68     var (mappings, declMap) = x.namespaces.foldLeft( // (nsmappings start with top , what we declare)
69       (currentMappings, Map[String, String]()))((p, a) => laddNS(a, p._1, p._2))
70 
71     // with our attributes
72     val t = x.attributes.foldLeft((mappings, declMap)) { (p, a) =>
73       val e = a
74       if (e.prefix.isDefined)
75         laddNS((e.prefix.get, e.namespace.uri), p._1, p._2)
76       else p
77     }
78 
79     mappings = t._1
80     declMap = t._2
81 
82     // is this namespace declared?
83     val addedDefaultNS = {
84       val currentDefault = mappings(NamespaceDefaults.namespace.uri)
85       if (x.name.namespace != NamespaceDefaults.noNamespace) {
86         // is there a prefix?
87         if (x.name.prefix.isDefined) {
88           // then it is a normal laddNS
89           val (imappings, ideclMap) = laddNS((x.name.prefix.get, x.name.namespace.uri), mappings, declMap)
90           mappings = imappings
91           declMap = ideclMap
92           // push the current on again
93           false
94         } else {
95           // there will always be on as we don't remove only redefine.
96 
97           // is it the current default namespace? 
98           if (currentDefault == x.name.namespace.uri) {
99             // keep the current
100             false
101           } else {
102             // must redefine it
103             // then the namespace has been badly declared as we have a conflict, but in order for at
104             // least this element to be valid we must remap it, log it?
105             mappings = mappings.updated(NamespaceDefaults.namespace.uri, x.name.namespace.uri)
106             true
107           }
108         }
109       } else {
110         // then if the current default is not "" we must redeclare it
111         if (currentDefault == NamespaceDefaults.noNamespace.uri) {
112           false
113         } else {
114           mappings = mappings.updated(NamespaceDefaults.namespace.uri, NamespaceDefaults.noNamespace.uri)
115           true
116         }
117       }
118     }
119 
120     // don't render it here, mappings contains it
121     declMap = declMap - NamespaceDefaults.namespace.uri
122 
123     val default = x.namespaces.get(NamespaceDefaults.namespace.uri)
124     val addDef =
125       if (addedDefaultNS || default.isDefined) {
126         // we have a new mapping, old one doesn't work anymore
127         Some(mappings(NamespaceDefaults.namespace.uri))
128       } else
129         None
130 
131     NamespaceContext(mappings, declMap, addDef)
132   }
