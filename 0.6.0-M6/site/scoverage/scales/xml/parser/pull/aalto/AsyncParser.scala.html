210   /**
211    * Given a DataChunk will return the next available data stream.
212    * Returns EOF when the Parser is closed, Empty when more DataChunks are needed.
213    * Returning El[EphemeralStream[PullType]] provides a stream which will lazily evalulate all available Xml Events from this DataChunk.
214    *
215    *
216    * NOTEs (see Aalto-xml feeder.feedInput for details on this):
217    * 1) All existing events from the last El should be consumed before calling nextInput again
218    * 2) The DataChunk passed in should not be re-used until the stream returns empty (or Empty/EOF is returned).
219    */ 
220   def nextInput(d: DataChunk): Input[EphemeralStream[PullType]] = {
221     //println("called nextInput")
222 
223     if (d.isEmpty) {
224       //println("fed empty")
225       Input.Empty[EphemeralStream[PullType]]
226     } else 
227       if (isClosed || d.isEOF) {
228         if (d.isEOF) {
229           //println("Data was eof")
230           closeResource
231         }
232         Eof[EphemeralStream[PullType]]
233       } else {
234         if (!feeder.needMoreInput) {
235           error("The stream from the previous call to nextInput was not evaluated ")
236         }
237 
238         //println("pushing "+d)
239         feeder.feedInput(d.array, d.offset, d.length)
240 
241         val res = nextStream()
242 
243         if (res.isEmpty) { // let the enumerator deal with it
244           //println("empty after feeding")
245           Input.Empty[EphemeralStream[PullType]]
246         } else // it may have empty after the call
247           Element[EphemeralStream[PullType]](res)
248     }
