367   def text(implicit t: XmlTree): String = {
368     t.fold(new StringBuilder()) 
369     { (walker, sb) =>
370       if (walker.isLeft) {
371         walker.left.get match {
372           case Text(text) => sb.append(text)
373 	        case CData(text) => sb.append(text)
374           case _ => ()
375         }
376       }
377      sb
378    }.toString
379   }
