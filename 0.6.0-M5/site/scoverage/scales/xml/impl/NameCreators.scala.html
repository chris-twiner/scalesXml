7 /**
8  * Given an optimisation strategy create QNames for elements and attributes.

10  * These are primarily of use from parsers and handle 0 length uris and qNames with ":" in them.

14   /** Elems can have any QName, attribs only prefixed or default */
15   def eqn[Token <: OptimisationToken](uri: String,
16     localName: String,
17     qName: String, strategy : MemoryOptimisationStrategy[Token], token : Token): QName =
18     if (uri.length() == 0) {
19       // can only be in qName
20       strategy.noNamespaceQName(qName, token)
21     } else {
22       // might have a prefix
23       if (qName == localName) {
24         // no prefix
25 	strategy.unprefixedQName(localName, uri, token)
26       } else {
27         // break up the prefix
28         val bits = qName.split(':')
29 	
30 	strategy.prefixedQName(localName, uri, bits(0), token)
31       }
32     }
33 
34   /** attribs only prefixed or default */
35   def aqn[Token <: OptimisationToken](uri: String,
36     localName: String,

38     if (uri.length() == 0) {
39       // can only be in qName
40       strategy.noNamespaceQName(qName, token) // Right
41     } else {
42       // might have a prefix

44         // no prefix
