196   def pumpEvent[Token <: OptimisationToken](parser: XMLStreamReader,
197                                             strategy: MemoryOptimisationStrategy[Token],
198                                             token: Token,
199                                             prolog: Prolog,
200                                             idepth: Int,
201                                             strictPath: List[QName] = Nil,
202                                             ipath: List[QName]= Nil)(otherEventHandler: Either[Int, Throwable] => PullType) : (PullType, Int, Int, Prolog, List[QName]) = {
203     var depth = idepth
204     var vprolog = prolog
205     var vpath = ipath
206 
207     @tailrec
208     def dropWhile(): PullType = {
209       def doStep(event: Int): Option[PullType] = {
210         event match {
211           case XMLStreamConstants.START_ELEMENT =>
212             depth += 1
213             val elemQName: QName = PullUtils.getElemQName(parser, strategy, token)
214             vpath = vpath :+ elemQName
215             if (vpath.equals(strictPath)) {
216               val attributes = PullUtils.getAttributes(parser, strategy, token)
217               val namespaces = PullUtils.getNamespaces(parser, strategy, token)
218               Some(strategy.elem(elemQName, attributes, namespaces, token))
219             } else {
220               None
221             }
222           case XMLStreamConstants.END_ELEMENT =>
223             depth -= 1
224             val isMissingStrictPath = vpath.equals(strictPath.take(strictPath.size - 1))
225             if (isMissingStrictPath) {
226               val elemQName = PullUtils.getElemQName(parser, strategy, token)
227               val namespaces = PullUtils.getNamespaces(parser, strategy, token)
228               vpath = ipath.take(ipath.size - 1)
229               Some(EndElem(elemQName, namespaces))
230             } else {
231               vpath = vpath.take(vpath.size - 1)
232               None
233             }
234           case XMLStreamConstants.END_DOCUMENT =>
235             Some(null)
236           case _ =>
237             None
238         }
239       }
240 
241       if (parser.hasNext) {
242         doStep(parser.next()) match {
243           case Some(res) => res
244           case None => dropWhile()
245         }
246       } else {
247         error("this should never happen")
248       }
249     }
250 
251     var nextEvent = XMLStreamConstants.END_DOCUMENT // use this in the case of error from calling next as well, blow it up but try to shut down
252 
253     val event: PullType =
254       try {
255         nextEvent = parser.next
256 
257         nextEvent match {
258           case XMLStreamConstants.START_ELEMENT => //1
259             depth += 1
260             if (strictPath.isEmpty)
261               strategy.elem(getElemQName(parser, strategy, token), getAttributes(parser, strategy, token), getNamespaces(parser, strategy, token), token)
262             else {
263               val elemQName = PullUtils.getElemQName(parser, strategy, token)
264               vpath = ipath :+ elemQName
265               val validSubtree = vpath.take(strictPath.size).equals(strictPath)
266               if (idepth == StartDepth || validSubtree) {
267                 val attributes = PullUtils.getAttributes(parser, strategy, token)
268                 val namespaces = PullUtils.getNamespaces(parser, strategy, token)
269                 strategy.elem(elemQName, attributes, namespaces, token)
270               } else {
271                 dropWhile()
272               }
273             }
274           case XMLStreamConstants.END_ELEMENT => //2
275             depth -= 1
276             if (strictPath.isEmpty)
277               EndElem(getElemQName(parser, strategy, token), getNamespaces(parser, strategy, token))
278             else {
279               val elemQName = PullUtils.getElemQName(parser, strategy, token)
280               val namespaces = PullUtils.getNamespaces(parser, strategy, token)
281               vpath = ipath.take(ipath.size - 1)
282               EndElem(elemQName, namespaces)
283             }
284           case XMLStreamConstants.CHARACTERS => Text(parser.getText)
285           case XMLStreamConstants.CDATA => CData(parser.getText)
286           case XMLStreamConstants.COMMENT => Comment(parser.getText)
287           case XMLStreamConstants.PROCESSING_INSTRUCTION => PI(parser.getPITarget(), parser.getPIData())
288           case XMLStreamConstants.SPACE => Text(parser.getText) // jdk impl never calls but to be safe we should grab it
289           case XMLStreamConstants.START_DOCUMENT => {
290             // get the encoding etc
291             // NB the asynch variety can also call this, if no more events are available then it returns the waiting object.
292             val ec = parser.getCharacterEncodingScheme()
293 
294             vprolog = vprolog.copy(decl = Declaration(
295               version = if (parser.getVersion() == "1.1")
296                 Xml11 else Xml10,
297               encoding = if (ec eq null) defaultCharset else java.nio.charset.Charset.forName(ec), // TODO what do we do about unsupported, throwing is probably fine, but it irritates, if we can get here the parser at least supports it, even if we can't write to it
298               standalone = parser.isStandalone()))
299 
300             val (nev, nex, nde, vvp, _) = pumpEvent(parser, strategy, token, vprolog, depth)(otherEventHandler) // we don't want to handle this
301 
302             // reset to keep the correct values
303             nextEvent = nex
304             depth = nde
305             vprolog = vvp
306             nev
307           }
308           case XMLStreamConstants.DTD => dtdDummy // push it through in start
309 
310           // we don't really want to handle other types?
311           case _ => otherEventHandler(Left(nextEvent))
312         }
313       } catch {
314         case t: Throwable => otherEventHandler(Right(t))
315       }
316     (event, nextEvent, depth, vprolog, vpath)
317   }
