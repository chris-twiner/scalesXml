68 /**
69  * Collection of optimisations that reduce memory significantly at the cost of parsing performance, and later non-lazy creation of correct objects for accessing the tree.
70  *
71  * This can have confusing runtime evaluation of XPaths etc.  These optimisations are appropriate for trees that are seldom read, repeated reads may generate too much garbage.
72  */ 
73 object LazyOptimisedTree {
74 
75   /**
76    * Returns an optimised Tree if Possible
77    */ 
78   def apply( section : Elem, children : XmlChildren ) ( implicit fromParser : FromParser ) : XmlTree = 
79     if (children.isEmpty)
80       Tree(section, children)
81     else {
82       val head = children.head
83       if (head.isLeft) {
84 	val left = head.getLeft
85 	if (left.isInstanceOf[Text] && children.size == 1) {
86 	  if (section.attributes.isEmpty && section.namespaces.isEmpty)
87 	    newNameValue(section.name, left.value)
88 	  else
89 	    new ElemValue(section, left.value)
90 	} else Tree(section, children)
91       } else Tree(section, children)
92     }
93   
94 
95   def newNameValue( iname : QName, itext : String ) ( implicit fromParser : FromParser, iseqLikeThing: SeqLikeThing[XCC[_], ItemOrTree[XmlItem, Elem, XCC], XCC] ) : NameValue =
96     if (fromParser eq NotFromParser)
97       new NameValue(iname, itext) {
98 	
99         def section : Elem = Elem(name)(NotFromParser) // have to recheck that the qname is ok
100 
101         override implicit val seqLikeThing: SeqLikeThing[XCC[_], ItemOrTree[XmlItem, Elem, XCC], XCC] = iseqLikeThing
102       }
103     else
104       new NameValue(iname, itext) {
105 	
106 	      def section : Elem = Elem(name)(IsFromParser) // don't have to recheck
107 
108         override implicit val seqLikeThing: SeqLikeThing[XCC[_], ItemOrTree[XmlItem, Elem, XCC], XCC] = iseqLikeThing
109       }
