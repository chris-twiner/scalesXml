23   /**
24    * The serializer will be returned automatically to the pool by calling closer
25    *
26    * doc functions are only evaluated upon the first elem / last elem
27    */
28   def serializeIter[F[_]]( output : XmlOutput, serializer : Serializer, closer : () => Unit, doc : DocLike = EmptyDoc())(implicit F: Monad[F]) : SerialIterT[F] = {
29     // cannot pass it on the stack as we'll get odd ordering with trampolines
30     var status: StreamStatus = StreamStatus(output)
31 
32     var prevE: PullType = null
33 
34     def shouldPump(e: PullType): Option[PullType] =
35       if (e ne prevE) {
36         prevE = e
37         Some(e)
38       } else
39         // Duplicates happen when restarting the processing in the face of trampolines.  Using run without asynchronous empties does not suffer this issue
40         None
41 
42 
43     var empties = 0
44 
45     def done(status: StreamStatus): F[SerialStepT[F]] = {
46       // give it back
47       closer()
48       //println("empties was "+empties)
49       F.point(Done((status.output, status.thrown), Eof[PullType]))
50     }
51 
52     def go(serializer: Serializer)(s: Input[PullType]): SerialIterT[F] =
53       iterateeT(
54         s(el = e => {
55           if (status.thrown.isDefined) done(status)
56           else {
57             val r = shouldPump(e)
58             F.point(
59               r.fold(Cont(go(serializer))) { e =>
60                 val nstatus =
61                   if (!status.haveSetProlog) {
62                     // decl and prolog misc, which should have been collected by now
63                     val opt = serializer.xmlDeclaration(status.output.data.encoding,
64                       status.output.data.version).orElse {
65                       serializeMisc(status.output, doc.prolog.misc, serializer)._2
66                     }
67 
68                     status.copy(thrown = opt, haveSetProlog = true)
69                   } else
70                     status
71 
72                 val r = StreamSerializer.pump(e, nstatus, serializer)
73                 status = r
74 
75                 //println("pumped rest e " + System.identityHashCode(e) + " r " + System.identityHashCode(r))
76                 Cont(go(serializer))
77               }
78             )
79           }
80         },
81           empty = {
82             empties += 1
83             //println("outitr empty " +System.identityHashCode(prev))
84             F.point(Cont(go(serializer)))
85           },
86           eof = {
87             if (status.thrown.isDefined) done(status)
88             else {
89               val r = StreamSerializer.pump(StreamSerializer.EOF, status, serializer)
90               val opt = serializeMisc(r.output, doc.end.misc, serializer)._2
91 
92               val lastStatus = r.copy(thrown = opt)
93 
94               done(lastStatus)
95             }
96           }
97         )
98       )
99 
100     iterateeT(F.point(Cont(go(serializer))))
101   }

121       }
122     }
