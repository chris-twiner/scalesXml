29   /**
30    * Pushes a single PullType to the given output, requires a finishing "dummy" event to indicate the end of the document.
31    * If status.thrown is defined status is returned
32    */
33   def pump(currentEvent: PullType, status : StreamStatus, serializer: Serializer) : StreamStatus = {
34     // shadow it
35     import status.output.{serializerF => defaultSerializerFactory}
36     import status.{output, isEmpty}
37 
38     if (status.thrown.isDefined) status
39     else {
40 
41       (status.prev, currentEvent) match {
42         case (None, EOF) =>
43           status.copy(thrown = Some(NoDataInStream()))
44         case (None, e) =>
45           status.copy(prev = Some(currentEvent))
46         case (Some(Right(lastElem)), EOF) if status.isEmpty =>
47           status // nothing to do
48         case (Some(Right(lastElem)), EOF) =>
49           status.copy(output.copy(currentMappings = output.currentMappings.tail,
50             path = output.path.tail),
51             serializer.endElement(lastElem.name, output.path), false, Some(currentEvent))
52         case (Some(Left(p: XmlItem)), _) => // allow EOF's through for xmlitems
53           status.copy(output, serializer.item(p, output.path), false, Some(currentEvent))
54         case (_, EOF) =>
55           status.copy(thrown = Some(new IllegalStateException("EOF should only be sent after the closing element")))
56         case (Some(Right(endElem)), Left(i: XmlItem)) if !status.isEmpty => // don't do this for other end elements
57           status.copy(output.copy(currentMappings = output.currentMappings.tail,
58               path = output.path.tail),
59               serializer.endElement(endElem.name, output.path), false, Some(currentEvent))
60         case (Some(Right(endElem)), Left(i: XmlItem)) if status.isEmpty => // it's already closed
61           status.copy(prev = Some(currentEvent))
62         case (Some(Right(endElem)), Left(x: Elem)) if !status.isEmpty => // unrelated elements
63           status.copy(output.copy(currentMappings = output.currentMappings.tail,
64               path = output.path.tail),
65               serializer.endElement(endElem.name, output.path), false, Some(currentEvent))
66         case (Some(Left(x: Elem)), Right(endElem)) =>
67           // must be an empty
68           // x.namespaces can't be used any further
69           val nc = doElement(x, output.currentMappings.head)
70           status.copy(output, serializer.emptyElement(x.name, x.attributes, nc.declMap, nc.addDefault, x.name :: output.path), true, Some(currentEvent)) // let us know to ignore the next end
71         case (Some(Left(x: Elem)), _) =>
72           val npath = x.name :: output.path
73 
74           val nc = doElement(x, output.currentMappings.head)
75           status.copy(output.copy(currentMappings = nc.mappings +: output.currentMappings, path = npath),
76             serializer.startElement(x.name, x.attributes, nc.declMap, nc.addDefault, npath), false, Some(currentEvent))
77       }
78     }
79   }

83  * Provides a base class for steam handling
84  */
85 class StreamSerializer[T](toP : T => Iterator[PullType]) extends SerializeableXml[T] {
86   import StreamSerializer._
87 
88   /**
89    * Override to provide an actual doc
90    */
91   def doc(it: T) : DocLike = EmptyDoc()
92   def apply(itT: T)(out: XmlOutput, serializer: Serializer): (XmlOutput, Option[Throwable]) = {
93     // 2.10 forces implicits at copy use site NOT at declaration site
94     import out.serializerF
95 
96     val it = toP(itT)
97     // left of the sequence is our actual,
98     val allButLast = it.foldLeft(StreamStatus(out)) {  (status, next) =>
99       if (status.thrown.isDefined) status
100       else {
101         pump(next, status, serializer)
102       }
103     }
104     // complete it
105     val r = pump(EOF, allButLast, serializer)
106     (r.output, r.thrown)
107   }
108 }
