29    * Pushes a single PullType to the given output, requires pairing of events.
30    * If status.thrown is defined status is returned
31    */ 
32   def pump(event: (PullType, PullType), status : StreamStatus, serializer: Serializer) : StreamStatus = {
33     // shadow it
34     import status.output.{serializerF => defaultSerializerFactory}
35     import status.{output, isEmpty}
36 
37     if (status.thrown.isDefined) status
38     else {
39       
40       val (ev, next) = event
41       ev match {
42 	case Left(i: XmlItem) =>
43 	  StreamStatus(output, serializer.item(i, output.path), false)
44 	case Left(x: Elem) =>
45 	  // if next is an end elem then its an empty
46 	  if (next.isRight) {
47 
48 	    // x.namespaces can't be used any further
49 	    val nc = doElement(x, output.currentMappings.head)
50 	    StreamStatus(output, serializer.emptyElement(x.name, x.attributes, nc.declMap, nc.addDefault, x.name :: output.path), true) // let us know to ignore the next end
51 	  } else {
52 	    val npath = x.name :: output.path
53 
54 	    val nc = doElement(x, output.currentMappings.head)
55 	    StreamStatus(output.copy(currentMappings = nc.mappings +: output.currentMappings, path = npath),
56 	      serializer.startElement(x.name, x.attributes, nc.declMap, nc.addDefault, npath), false)
57 	  }
58 	case Right(endElem) =>
59 	  if (isEmpty)
60 	    StreamStatus(output, None, false)
61 	  else
62 	    // pop the last ones
63 	    StreamStatus(output.copy(currentMappings = output.currentMappings.tail,
64 				     path = output.path.tail), 
65 	      serializer.endElement(endElem.name, output.path), false)
66 
67       }
68     }
69   }
