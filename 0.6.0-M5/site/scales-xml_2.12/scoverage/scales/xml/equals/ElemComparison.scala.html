22 /**
23  * Namespaces by default are not compared.  They aren't marked as implicit as you really have to be explicitly wanting to compare them.
24  *
25  * An Attributes comparison and a QName Equal instance are required.  This also allows, for example, specification of equality using prefixes for the Elem, but not on the Attributes.
26  */ 
27 class ElemComparison(namespaces : Equal[Map[String, String]] = ElemEqualHelpers.allwaysTrueNamespacesEqual)( implicit ac : XmlComparison[Attributes], eqn : Equal[QName]) extends XmlComparison[Elem] {
28   def compare( calculate : Boolean, context : ComparisonContext, left : Elem, right : Elem) : Option[(XmlDifference[_], ComparisonContext)] = {
29     if (left eq right)
30       None
31     else
32     if (!(left.name === right.name))
33       if (calculate)
34 	Some((ElemNameDifference(left,right), context))
35       else
36 	noCalculation
37       else 
38 	ac.compare(calculate, context, left.attributes, right.attributes).
39     cata( x => 
40       if (calculate) {
41 	val (attrs : AttributesDifference, p) = x
42 	Some( (ElemAttributeDifference( left, right, attrs), context) )
43       } else noCalculation
44     ,
45       if (!namespaces.equal(left.namespaces, right.namespaces))
46 	if (calculate)
47 	  Some( (ElemNamespacesDifference(left, right), context) )
48 	else
49 	  noCalculation
50 	else None
51     )
52   }
53 }
