72   /**

74    *
75    * Use raw to convert without forcing sorting or checking for duplicates.
76    *
77    * NOTE Any laziness aquired by viewed will also be lost by the flatten
78    */
79   def raw[T <: Iterable[XmlPath]](xpath : XPath[T]) : Iterable[XmlPath] =
80     xpath.path.nodes.flatten
81 
82   /**
83    * Normally the implicit fromXPathToIterable will provide sorted output, but this can create significant slow down when only the contents are necessary, not their order.
84    *
85    * Use lazyRaw to convert without forcing sorting or checking for duplicates, and to evaluate lazily.
86    *
87    * Warning, don't use this version of raw unless your XPath usage costs more than the navigation, its based on Stream and terribly slow (huge memory and stack usage etc).
88    *
89    * NOTEs
90    * Laziness with flatten is achieved by using iterator.toIterable before the flatten (uses Stream internally).
91    * Its only useful as a lazy evaluator if viewed was used.
92    */
93   def lazyRaw[T <: Iterable[XmlPath]](xpath : XPath[T]) : Iterable[XmlPath] =
94     xpath.path.nodes.iterator.toStream.flatten
