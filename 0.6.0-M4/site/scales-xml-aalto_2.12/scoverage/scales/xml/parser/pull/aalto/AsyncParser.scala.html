201   /**
202    * Given a DataChunk will return the next available data stream.
203    * Returns EOF when the Parser is closed, Empty when more DataChunks are needed.
204    * Returning El[EphemeralStream[PullType]] provides a stream which will lazily evalulate all available Xml Events from this DataChunk.
205    *
206    *
207    * NOTEs (see Aalto-xml feeder.feedInput for details on this):
208    * 1) All existing events from the last El should be consumed before calling nextInput again
209    * 2) The DataChunk passed in should not be re-used until the stream returns empty (or Empty/EOF is returned).
210    */ 
211   def nextInput(d: DataChunk): Input[EphemeralStream[PullType]] = {
212     //println("called nextInput")
213 
214     if (d.isEmpty) {
215       //println("fed empty")
216       Input.Empty[EphemeralStream[PullType]]
217     } else 
218       if (isClosed || d.isEOF) {
219         if (d.isEOF) {
220           //println("Data was eof")
221           closeResource
222         }
223         Eof[EphemeralStream[PullType]]
224       } else {
225         if (!feeder.needMoreInput) {
226           error("The stream from the previous call to nextInput was not evaluated ")
227         }
228 
229         //println("pushing "+d)
230         feeder.feedInput(d.array, d.offset, d.length)
231 
232         val res = nextStream()
233 
234         if (res.isEmpty) { // let the enumerator deal with it
235           //println("empty after feeding")
236           Input.Empty[EphemeralStream[PullType]]
237         } else // it may have empty after the call
238           Element[EphemeralStream[PullType]](res)
239 
240     }
