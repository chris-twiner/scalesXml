18   protected def add(path: Path[Item, Section, CC], direction: Int, newPath: Iterable[ItemOrTree[Item, Section, CC]]) : FoldR[Item, Section, CC] = {
19     // need to go up to replace
20     val parent = path.zipUp
21     if (path.top.isLeft && !rootChangeAllowed)
22       Right(AddedBeforeOrAfterRoot)
23     else
24       Left(parent.
25         modify { x =>
26           val tree = x.right.get;
27           val index = path.node.index + direction
28           val (pre,pos) = seqLikeThing.splitAt(tree.children)(index)
29           val newChildren = seqLikeThing.++((seqLikeThing.++(pre)(newPath)))( seqLikeThing.seq(pos))
30           Tree(tree.section, newChildren.asInstanceOf[CC[ItemOrTree[Item, Section, CC]]]) // actually the same
31         })
32 
33   }

84 /**
85  * Allows foldPositions to be nested, only replace and delete makes sense here (afaict).
86  *
87  * As such, when wholeTree is false, the path (which must be a tree) is transformed
88  * 
89  *     p => top(p.tree)
90  *
91  * and a special case for "deletes" is made - when RemovedRoot is returned from the transformation a delete will take place on the Path.  This enforces that only replace and removes are possible and function appropriately.
92  *
93  * Warning:
94  * 
95  * When wholeTree is true the function f is passed the Path (or item) in the original tree, any transformations are then conusmed across the whole tree, which is likely not desired. 
96  */
97 case class ReplaceWith[Item <: LeftLike[Item, Tree[Item, Section, CC]], Section, CC[_]](f: PathFoldR[Item, Section, CC], wholeTree : Boolean = false)
98   (implicit val seqLikeThing: SeqLikeThing[CC[_], ItemOrTree[Item, Section, CC], CC]) extends FoldOperation[Item, Section, CC] {
99 
100   def perform(path: Path[Item, Section, CC]): FoldR[Item, Section, CC] =
101     // modify back in (allows changes), or pass on the error
102     f( if (wholeTree) 
103       	 path
104        else
105          top(path.tree) ).
106             fold(fres =>
107                 Left(path.modify(_ =>
108                         fres.tree
109                   )),
110               x => {
111                 if ((x eq RemovedRoot) &&
112                     (!wholeTree))
113                   // delete the node - special case as per doc
114                   Remove[Item, Section, CC]().perform(path) // it might itself be the root but thats fine
115                 else
116                   Right(x)
117               }
118             )
119 
120 }
