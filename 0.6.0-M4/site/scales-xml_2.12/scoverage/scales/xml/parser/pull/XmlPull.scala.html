65     val token = strategy.createToken(Xml10, IsFromParser)
66     // start with nothing
67     val buf = new impl.TreeProxies()
68 
69     while( pull.hasNext ){
70       pull.next match {
71 	case Left( i : XmlItem ) => 
72 	  buf.addChild(i)
73 	case Left( e : Elem ) => 
74 	  strategy.beginSubTree(buf, e, token)
75 	case Right(endElem) => 
76 	  strategy.elementEnd(buf, token)
77       }
78     }
79     buf.tree
80   }
