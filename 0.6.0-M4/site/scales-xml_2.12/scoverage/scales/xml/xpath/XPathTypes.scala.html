141    */ 
142   def eager_recUnpack(xmlPath: Iterable[XmlPath]): List[XmlPath] = {
143     var res : List[XmlPath] = Nil
144 
145     val i = xmlPath.iterator
146     while(i.hasNext) {
147       val child = i.next
148       if (child.isItem) 
149 	res = child :: res
150       else {
151 	// child first then its kids
152 	res = (child :: res ).reverse_::: (
153 	  eager_recUnpack(child: Iterable[XmlPath]) )
154       }
155     }
156     res
157   }
